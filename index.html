üî∏ Lists as Stacks and Queues

Lists can be used as both stacks (LIFO) and queues (FIFO) data structures

# Using list as a Stack (Last-In-First-Out)
stack = []
stack.append(1)        # Push ‚Üí [1]
stack.append(2)        # Push ‚Üí [1, 2]
stack.append(3)        # Push ‚Üí [1, 2, 3]
last = stack.pop()     # Pop 3 ‚Üí [1, 2]
print(stack[-1])       # Peek at top: 2
# Using list as a Queue (First-In-First-Out)
from collections import deque
queue = deque([])
queue.append(1)        # Enqueue ‚Üí [1]
queue.append(2)        # Enqueue ‚Üí [1, 2]
queue.append(3)        # Enqueue ‚Üí [1, 2, 3]
first = queue.popleft() # Dequeue 1 ‚Üí [2, 3]
print(queue[0])        # Peek at front: 2
# Performance considerations
# Stack operations (append/pop) are O(1)
# Queue with list.pop(0) is O(n), use deque instead!
queue = deque(maxlen=3)  # Optional size limit
queue.extend([1, 2, 3])  # Bulk enqueue
queue.rotate(1)          # Rotate right
queue.rotate(-1)         # Rotate left

class Temperature:
    def __init__(self, celsius):
        self._celsius = celsius  # Protected attribute
    
    @property
    def fahrenheit(self):
        return (self._celsius * 9/5) + 32

temp.fahrenheit = 32
print(temp._celsius)     # 0.0

        @functools.wraps(func)

                return cache[args]
            result = func(*args)
            cache[args] = result
            return result

        # Get keyboard input

        self.direction.x = keys[pygame.K_RIGHT] - keys[pygame.K_LEFT]
        self.direction.y = keys[pygame.K_DOWN] - keys[pygame.K_UP]
        
        # Normalize for diagonal movement
        if self.direction.length() > 0:
            self.direction = self.direction.normalize()
        
        # Move the player
        self.rect.x += self.direction.x * self.speed
        self.rect.y += self.direction.y * self.speed
